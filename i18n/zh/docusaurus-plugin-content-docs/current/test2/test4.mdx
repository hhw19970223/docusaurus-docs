```json

      {
        "language": "en",
        "tabs": [
          {
            "tab": "Docs",
            "groups": [
              {
                "group": "Get Started",
                "pages": [
                  "en/docs/introduction",
                  "en/docs/quickstart",
                  "en/docs/development"
                ]
              },
              {
                "group": "Essentials",
                "pages": [
                  "en/docs/normal/markdown",
                  "en/docs/normal/code",
                  "en/docs/normal/images",
                  "en/docs/normal/settings",
                  "en/docs/normal/navigation",
                  "en/docs/normal/reusable-snippets"
                ]
              }
            ]
          },
          {
            "tab": "API",
            "groups": [
              {
                "group": "API Documentation",
                "pages": [
                  "en/api/introduction"
                ]
              },
              {
                "group": "Endpoint Examples",
                "pages": [
                  "en/api/endpoint/get",
                  "en/api/endpoint/create",
                  "en/api/endpoint/delete",
                  "en/api/endpoint/webhook"
                ]
              }
            ]
          }
        ]
      }
    ]
```
